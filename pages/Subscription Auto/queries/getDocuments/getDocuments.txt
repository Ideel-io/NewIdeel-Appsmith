query GetDocuments(
  $userId: String!,
  $profileId: String!,
	$recurrentSubscriptionId: String!,
) {
	identityRecto: getDocumentsByLabel(
    userId: $userId,
    folderId: $profileId,
    label: "IDENTITY_RECTO"
  ) {
    fileName,
    downloadUrl,
    uploadDate
  }
  identityVerso: getDocumentsByLabel(
    userId: $userId,
    folderId: $profileId,
    label: "IDENTITY_VERSO"
  ) {
    fileName,
    downloadUrl,
    uploadDate
  }
  drivingLicense: getDocumentsByLabel(
    userId: $userId,
    folderId: $recurrentSubscriptionId,
    label: "DRIVING_LICENSE"
	) {
    fileName,
    downloadUrl,
    uploadDate
  }
  grayCard: getDocumentsByLabel(
    userId: $userId,
    folderId: $recurrentSubscriptionId,
    label: "GRAY_CARD"
	) {
    fileName,
    downloadUrl,
    uploadDate
  }
  informationStatement: getDocumentsByLabel(
    userId: $userId,
    folderId: $recurrentSubscriptionId,
    label: "INFORMATION_STATEMENT"
	) {
    fileName,
    downloadUrl,
    uploadDate
  }
	previousContract: getDocumentsByLabel(
    userId: $userId,
    folderId: $recurrentSubscriptionId,
    label: "PREVIOUS_CONTRACT"
	) {
    fileName,
    downloadUrl,
    uploadDate
  }
}
